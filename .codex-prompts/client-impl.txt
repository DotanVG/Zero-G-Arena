=== client/src/physics.ts ===
Import Vector3 from 'three', constants from shared, clampMagnitude from util/math.
interface PhysicsState { pos: THREE.Vector3, vel: THREE.Vector3 }
function integrate(state: PhysicsState, axes: {x:number,y:number,z:number}, forward: THREE.Vector3, right: THREE.Vector3, up: THREE.Vector3, boost: boolean, dt: number): void
  const accelDir = right.clone().multiplyScalar(axes.x)
    .add(up.clone().multiplyScalar(axes.y))
    .add(forward.clone().multiplyScalar(axes.z))
  state.vel.addScaledVector(accelDir, ACCEL * dt)
  if (boost) state.vel.addScaledVector(forward, BOOST)
  state.vel.multiplyScalar(DAMPING)
  clampMagnitude(state.vel, MAX_SPEED)
  state.pos.addScaledVector(state.vel, dt)

function bounceArena(state: PhysicsState): void
  const limit = ARENA_SIZE / 2 - PLAYER_RADIUS
  for each axis (x,y,z): if pos > limit: pos = limit, vel = -abs(vel)*0.5; if pos < -limit: same

Import * as THREE from 'three', PhysicsState/integrate/bounceArena from physics, InputManager from input, CameraController from camera, Arena from arena/arena, makePlayerMaterial from render/materials, FREEZE_TIME/RESPAWN_TIME/INVULN_TIME from shared/constants.
Class LocalPlayer:
- phys: PhysicsState = { pos: new Vector3(0,0,-15), vel: new Vector3() }
- team: 0|1 = 0
- score = 0
- state: 'ACTIVE'|'FROZEN'|'RESPAWNING' = 'ACTIVE'
- frozenTimer = 0, respawnTimer = 0, invulnTimer = 0, boostCooldown = 0
- private mesh: THREE.Mesh
- constructor(scene: THREE.Scene): create SphereGeometry(0.8,8,8), makePlayerMaterial(0), add to scene
- update(input: InputManager, cam: CameraController, arena: Arena, dt: number): void
    if ACTIVE:
      boost = input.isBoost() && boostCooldown<=0
      if boost: boostCooldown = 1.5
      boostCooldown = max(0, boostCooldown-dt)
      integrate(phys, input.getAxes(), cam.getForward(), cam.getRight(), cam.getUp(), boost, dt)
      bounceArena(phys)
      invulnTimer = max(0, invulnTimer-dt)
      for goal of arena.getGoalPlanes():
        if invulnTimer<=0 && goal.checkEntry(phys.pos, team): score++; respawn()
    if FROZEN:
      frozenTimer -= dt; phys.vel.multiplyScalar(0.7)
      if frozenTimer<=0: state='ACTIVE'
    if RESPAWNING:
      respawnTimer -= dt
      if respawnTimer<=0: state='ACTIVE'; invulnTimer=INVULN_TIME
- freeze(impulse: THREE.Vector3): void — state=FROZEN; frozenTimer=FREEZE_TIME; phys.vel.add(impulse)
- respawn(): void — state=RESPAWNING; respawnTimer=RESPAWN_TIME; phys.pos.set(0,0,team===0?-15:15); phys.vel.set(0,0,0)
- getPosition(): THREE.Vector3
- getMesh(): THREE.Mesh
- getFrozenTimer(): number
- getScore(): number

Import * as THREE from 'three', SceneManager, InputManager, CameraController, LocalPlayer, Arena, HUD.
Class App:
- private sceneMgr: SceneManager
- private input: InputManager
- private cam: CameraController
- private player: LocalPlayer
- private arena: Arena
- private hud: HUD
- private lastTime = 0
- constructor():
    sceneMgr = new SceneManager()
    input = new InputManager()
    cam = new CameraController(sceneMgr.getCamera())
    arena = new Arena(sceneMgr.getScene())
    player = new LocalPlayer(sceneMgr.getScene())
    hud = new HUD()
- start(): void
    hud.showStart()
    document.addEventListener('click', ()=>{ if(!input.isLocked()){ input.lockPointer(sceneMgr.getRenderer().domElement); hud.hideStart() } })
    requestAnimationFrame((t)=>this.loop(t))
- private loop(timestamp: number): void
    const dt = Math.min((timestamp - this.lastTime) / 1000, 0.05)
    this.lastTime = timestamp
    if (input.isLocked()):
      const {dx,dy} = input.consumeMouseDelta()
      const roll = input.isRoll()
      cam.applyMouseDelta(dx, dy, (roll.right?1:0)-(roll.left?1:0), input.mouseSensitivity)
      input.updateFireCooldown(dt)
      player.update(input, cam, arena, dt)
      cam.apply(player.getPosition())
      hud.update({team0:player.getScore(),team1:0}, player.state, player.getFrozenTimer())
    sceneMgr.render()
    requestAnimationFrame((t)=>this.loop(t))
